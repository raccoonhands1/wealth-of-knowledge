> [!theorem]
> $$
> \begin{gather}
> \text{let the functions } f \text{ and } \frac{\partial f}{\partial y} \text{ be continuous in some rectangle } \alpha<t<\beta, \gamma<y<\delta \text{ containing the point } (t_{0}, y_{0}) \\
> \text{then, in some interval } t_{0}-h<t<t_{0}+h \text{ contained in } \alpha<t<\beta \text{, there is a unique solution } \\
> y=\phi(t) \text{ of the initial value problem} \\
> y^\prime = f(t,y), y(t_{0}) = y_{0}
>
>
> \end{gather}
> $$

> [!summary]
>
